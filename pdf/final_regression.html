MC_validation1 <- function(input, ...){
  n <- dim(model1_data)[1]
  n1 = round(0.80*n)
  n2 = n - n1  
  set.seed(2353)
  B = 100
  TrainErr0 = NULL
  TestErr0 = NULL
  for (b in 1:B){
    flag = sort(sample(1:n, n1))
    train_data = input[flag,]
    test_data  = input[-flag,]
    fit <- lm(NObeyesdad ~ . -NObeyesdad, data = input)
    # locate the response
    col_index <- which(names(input) == "NObeyesdad")
    # training error
    Pred1 <- predict(fit, newdata= train_data[, -col_index])
    TrainErr0 <- rbind(TrainErr0, mean(Pred1 - train_data[, col_index])^2)
    # prediction and prediction error
    Pred2 <- predict(fit, newdata= test_data[, -col_index])  # NObeyesdad at col 14
    TestErr0 <- rbind(TestErr0, mean(Pred2 - test_data[, col_index])^2)
  }
  return(list(train_errs = TrainErr0, test_errs = TestErr0, 
              av_train_err = mean(TrainErr0),
              av_test_err = mean(TestErr0)))
}
